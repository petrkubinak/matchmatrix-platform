services:
  postgres:
    image: postgres:16
    container_name: matchmatrix_postgres
    environment:
      POSTGRES_DB: matchmatrix
      POSTGRES_USER: matchmatrix
      POSTGRES_PASSWORD: matchmatrix_pass
    ports:
      - "5432:5432"
    volumes:
      - pg_data:/var/lib/postgresql/data
      # DDL + seed se spustí při prvním vytvoření volume (fresh DB)
      - ./db/05_DDL_postgres_v1.sql:/docker-entrypoint-initdb.d/05_DDL_postgres_v1.sql:ro
      - ./db/seed_v1.sql:/docker-entrypoint-initdb.d/10_seed_v1.sql:ro
    healthcheck:
      test: ["CMD-SHELL", "pg_isready -U matchmatrix -d matchmatrix"]
      interval: 5s
      timeout: 3s
      retries: 30

  redis:
    image: redis:7
    container_name: matchmatrix_redis
    ports:
      - "6379:6379"
    volumes:
      - redis_data:/data
    command: ["redis-server", "--appendonly", "yes"]
    healthcheck:
      test: ["CMD", "redis-cli", "ping"]
      interval: 5s
      timeout: 3s
      retries: 30

  pg_data:
